function addExpense(sheet, params) { var newRow = [ params.id, params.date, params.category, params.description, parseFloat(params.amount), params.paidBy, params.status, params.receiptRef ]; sheet.appendRow(newRow); return { success: true, message: "Expense saved" }; } function updateStatus(sheet, params) { var data = sheet.getDataRange().getValues(); for (var i = 1; i < data.length; i++) { if (data[i][0] && data[i][0].toString() === params.id) { sheet.getRange(i + 1, 7).setValue(params.status); break; } } return { success: true, message: "Status updated" }; } function deleteExpense(sheet, params) { var data = sheet.getDataRange().getValues(); for (var i = 1; i < data.length; i++) { if (data[i][0] && data[i][0].toString() === params.id) { sheet.deleteRow(i + 1); break; } } return { success: true, message: "Expense deleted" }; }
